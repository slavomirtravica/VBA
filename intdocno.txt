Sub IncrementAlpha()
Dim c As Range
For Each c In Range(Cells(1, 1), Cells(Rows.Count, 1).End(xlUp))
    If Not IsNumeric(Left(c, 1)) Then c.Offset(0, 1) = Chr(Asc(c) + 1)
Next c
End Sub
